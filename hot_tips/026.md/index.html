setup

wget https://github.com/hakimel/reveal.js/archive/master.tar.gz
tar -xzvf master.tar.gz
mv reveal.js-master reveal.js

convert

pandoc -t revealjs -s\
   -o myslides.html myslides.md \
   -V revealjs-url=https://unpkg.com/reveal.js@3.9.2/
